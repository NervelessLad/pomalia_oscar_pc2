cmake_minimum_required(VERSION 3.10)
project(AllOne_PC2)

set(CMAKE_CXX_STANDARD 17)

# Definimos la libreria core
add_library(allone_lib src/AllOne.cpp)
target_include_directories(allone_lib PUBLIC include)

# Crear el ejecutable de la demo
add_executable(demo demos/demo_allone.cpp)
target_link_libraries(demo allone_lib)

# Crear el ejecutable de las pruebas (CRÍTICO para tu evaluación)
add_executable(tests tests/tests.cpp)
target_link_libraries(tests allone_lib)

# Crear el ejecutable de benchmark
add_executable(benchmark_test benchmark/benchmark_allone.cpp)
target_link_libraries(benchmark_test allone_lib)
